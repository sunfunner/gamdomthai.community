คำสั่งจัดการไฟล์ทั่วไป
mkdir [Name] -- สร้าง Directory
rmdir -- ลบ Dirctory
cd = Change Directory
cd [Directory] -- ย้ายที่อยู่
pwd -- แสดง Directory ปัจจุบัน : Present working Directory
touch -- สร้างไฟล์
touch [Directory/Supdirectory/....] -- สร้างไฟล์แบบอิงที่ลงไฟล์
Faction > File -- ข้อมูลใส่ไฟล์ (ทับ)
Faction >> File -- เพิ่มข้อมูล
cat = concat
cat [File1] [File2] [...] -- เปิดไฟล์
rm -- ลบไฟล์

CTRL + C = หยุด Stop
CTRL + Z = พัก Pause
. -- ที่เดิม
.. -- ย้อนกลับ
* -- ทั้งหมด
ls -- แสดงลิส
ls -l -- แสดงลิส + รายละเอียด
ls -la -- แสดงลิสทั้งหมด
