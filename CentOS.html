Command Linux
adduser [Name] -- สร้าง User Account
passwd [Name] -- เซ็ต password ให้ User
usermod -aG [GroupName] [Name] -- ตั้ง Group ให้กับ User
logout -- ออกจากระบบ
cd /home
ls -- แสดงลิส
ls -l -- แสดงลิส + รายละเอียด
ls -la -- แสดงลิสรวมที่่อนไว้

sudo -- Super User Do
คำสั่ง โหลด ติดตั้ง
sudo yum install [Program] -- ลงโปรแกรม
sudo yum update -- อัพเดทโปรแกรม
pwd -- เปิด Directory (Folder นั่นแหละ) Present working Directory
mkdir [Name] -- สร้าง Directory
cd = Change Directory
CTRL + C = หยุด Stop
CTRL + Z = พัก Pause
