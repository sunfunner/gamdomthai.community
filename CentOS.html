Command Linux
adduser [Name] -- สร้าง User Account
passwd [Name] -- เซ็ต password ให้ User
usermod -aG [GroupName] [Name] -- ตั้ง Group ให้กับ User
logout -- ออกจากระบบ
cd /home
ls -- แสดงลิส
ls -l -- แสดงลิส + รายละเอียด
ls -la -- แสดงลิสรวมไว้

sudo -- Super User Do
คำสั่ง sudo dhclient -- สร้าง IP ต่อเน็ต
sudo yum install [Program] -- ลงโปรแกรม
sudo yum update -- อัพเดทโปรแกรม
pwd -- เปิด Directory (Folder นั่นแหละ) Present working Directory
mkdir [Name] -- สร้าง Directory
cd = Change Directory
cd [Directory] -- ย้ายที่อยู่
touch -- สร้างไฟล์
touch [Directory/Supdirectory/....]
Faction > File -- ข้อมูลใส่ไฟล์ (ทับ)
Faction >> File -- เพิ่มข้อมูล
cat = concat
cat [File1] [File2] -- เปิดไฟล์
. -- ที่เดิม
.. -- ย้อนกลับ
rm = Remove
rm -- ลบไฟล์
rmdir -- ลบ Dirctory
CTRL + C = หยุด Stop
CTRL + Z = พัก Pause
Text Editor Section
vi -- สร้าง Text File
กด i พิมข้อความ
กด ECS -- หยุดงานที่ทำใน Text File
:w -- Save ไงะ
:q -- ออกจากหน้า Text File
:q! -- ออกแบบไม่สนโลก
